from urllib.parse import urljoin
from lxml.html import fromstring
import re
URL = 'http://fantasy-h2h.ru/h2h/team_match_data/'
ITEM_PATH1 = '.breadcrumb li:first-child a'
ITEM_PATH2 = '.team_name'
ITEM_PATH3 = '.result a span'
ITEM_PATH4 = '.extra_goals'
ITEM_PATH5 = '.breadcrumb li:nth-child(5)'
def parse_courses():
 START = int(input("Enter START number: "))
 FINISH = int(input("Enter FINISH number: "))
 csvfile = open('2019result.csv', "w")
 i = START
 while i <= FINISH :
  adress = URL + str(i)
  f = urlopen(adress)
  list_html = f.read().decode('utf-8')
  list_doc = fromstring(list_html)
  print(i)
  if list_doc.cssselect(ITEM_PATH1)[0].text is None:
   i += 1
  else:
   csvfile.write(str(i) + '%')
   for elem in list_doc.cssselect(ITEM_PATH1):
    csvfile.write(elem.text + "%")
   for elem in list_doc.cssselect(ITEM_PATH5):
    try:
     csvfile.write(str(elem.text).split(", ")[1] + "%")
    except:
     csvfile.write(str('ошибка') + "%")		
   for elem in list_doc.cssselect(ITEM_PATH2):
    try:
     csvfile.write(elem.get('title') + "%")
    except:
     csvfile.write(str('ошибка') + "%")	
   for elem in list_doc.cssselect(ITEM_PATH3):
    csvfile.write(elem.text + "%")
   for elem in list_doc.cssselect(ITEM_PATH4):
    csvfile.write(elem.text + "%")   
   csvfile.write("\n")
   i += 1
 f.close()

 parse_courses()
if __name__ == '__main__':

Sub FindAndReplace()
  Dim MyRange As Range
  Dim MyCell As Range
  Set MyRange = Selection
  For Each MyCell In MyRange
    If Len(MyCell.Value) = 0 Then
      MyCell = 0
    End If
  Next MyCell
End Sub
